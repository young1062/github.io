# Classical Gram-Schmidt
classical_GS <- function(A) {
  m <- nrow(A)
  n <- ncol(A)
  
  Q <- matrix(0, m, n)
  R <- matrix(0, n, n)

# the special case of j=1
  R[1,1] <- sqrt(sum(A[,1]^2))
  Q[,1] <- A[,1]/R[1,1]
    
  for (j in 2:n) {
    vj <- A[, j]
    for (i in 1:(j-1)) {
      R[i, j] <- sum(Q[, i] %*% A[, j])
      vj <- vj - R[i, j] * Q[, i]
    }
    R[j, j] <- sqrt(sum(vj^2))
    Q[, j] <- vj / R[j, j]
  }
  
  return(list(Q = Q, R = R))
}


# Modified Gram-Schmidt
modified_GS <- function(A) {
  n <- ncol(A)
  Q <- matrix(0, nrow = nrow(A), ncol = n)
  R <- matrix(0, nrow = n, ncol = n)
  
  V <- A
  for (i in 1:n) {
    R[i,i] <- sqrt(sum(V[,i]^2))
    Q[,i] <- V[,i] / R[i,i]
    if (i < n){
    for (j in (i+1):n) {
      R[i,j] <- t(Q[,i]) %*% V[,j]
      V[,j] <- V[,j] - R[i,j] * Q[,i]
      }
    }
  }

  return(list(Q = Q, R = R))
}

The numerical example in class

eps <- 1e-10
a1 <- c(1, eps, 0, 0)
a2 <- c(1, 0, eps, 0)
a3 <- c(1, 0, 0, eps)

A <- cbind(a1, a2, a3)
result_CGS <- classical_GS(A)
result_MGS <- modified_GS(A)
print(result_CGS$Q)
##       [,1]       [,2]       [,3]
## [1,] 1e+00  0.0000000  0.0000000
## [2,] 1e-10 -0.7071068 -0.7071068
## [3,] 0e+00  0.7071068  0.0000000
## [4,] 0e+00  0.0000000  0.7071068
print(result_MGS$Q)
