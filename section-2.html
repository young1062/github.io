# Householder QR algorithm
householder_qr <- function(A) {
  m <- nrow(A)
  n <- ncol(A)
  R <- A
  Q <- diag(m)
  for(j in 1:n) {
    x <- R[j:m, j, drop=FALSE]
    e1 <- matrix(0, nrow=length(x), ncol=1)
    e1[1,1] <- 1
    v <- x + sign(x[1,1]) * norm(x, type="2") * e1
    v <- v / norm(v, type="2")
    R[j:m, j:n] <- R[j:m, j:n] - 2 * v %*% (t(v) %*% as.matrix(R[j:m, j:n]))
    Q[j:m, ] <- Q[j:m, ] - 2 * v %*% (t(v) %*% Q[j:m, ])
  }
  list(Q=t(Q), R=R)
}

# Generate Vandermonde Matrix
p <- seq(0.04, 1, length.out=25)
A <- outer(p, 0:14, function(p_i, j) p_i^j)

# Householder QR decomposition
result_house <- householder_qr(A)
Q_house <- result_house$Q
R_house <- result_house$R

# CGS QR
result_CGS <- classical_GS(A)
Q_CGS <- result_CGS$Q
R_CGS <- result_CGS$R

# MGS QR
result_MGS <- modified_GS(A)
Q_MGS <- result_MGS$Q
R_MGS <- result_MGS$R

# Error metrics
error_A_house <- norm(A - Q_house %*% R_house, type="2")
error_I_house <- norm(diag(25) - Q_house %*% t(Q_house), type="2")

error_A_CGS <- norm(A - Q_CGS %*% R_CGS, type="2")
error_I_CGS <- norm(diag(25) - Q_CGS %*% t(Q_CGS), type="2")

error_A_MGS <- norm(A - Q_MGS %*% R_MGS, type="2")
error_I_MGS <- norm(diag(25) - Q_MGS %*% t(Q_MGS), type="2")

print(paste0("A error of Householder:", error_A_house, "  CGS:", error_A_CGS, "  MGS:", error_A_MGS))
## [1] "A error of Householder:2.5690616987402e-15  CGS:4.32460657934178e-16  MGS:4.78604627250331e-16"
print(paste0("I error of Householder:", error_I_house, "  CGS:", error_I_CGS, "  MGS:", error_I_MGS))

The condition number of both \(A\) and \(A^TA\) are super large, suggesting that the QR decomposition on \(A\) may not be very stable. As for the \(||A-\tilde{Q} \tilde{R}||\), all of these three are doing a great job. However, when it comes to the orthogonality of \(Q\), householder beats the two Gram Schmidt algorithms. Among the two GM algorithms, Modified GS performs a little bit better.
